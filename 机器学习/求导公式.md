$$
(u \pm v)^{\prime}=u^{\prime} \pm v^{\prime} \quad(u v)^{\prime}=u^{\prime} v+u v^{\prime} \quad\left(\frac{u}{v}\right)^{\prime}=\frac{u^{\prime} v-u v^{\prime}}{v^{2}}
$$
